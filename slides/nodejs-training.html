Langkah 1

Model Mongoose untuk Aduan.

const aduanSchema = new mongoose.Schema({
  noAduan: { type: String, unique: true },
  namaPengadu: { type: String, required: [true, "Nama wajib diisi"] },
  kategori: {
    type: String,
    enum: { values: KATEGORI, message: 'Kategori "{VALUE}" tidak sah' },
  },
  status: { type: String, enum: STATUS, default: "Baru" },
}, { timestamps: true });

module.exports = mongoose.model("Aduan", aduanSchema);
MongooseSkema = struktur + validasi + nilai lalai
Langkah 2

Controller: Cipta & Baca.

// Baca + carian
exports.index = async (req, res) => {
  const { carian } = req.query;
  const pertanyaan = {};
  if (carian) {
    pertanyaan.$or = [
      { namaPengadu: { $regex: carian, $options: "i" } },
      // ...boleh tambah noAduan, premis
    ];
  }
  const senaraiAduan = await Aduan.find(pertanyaan).sort({ createdAt: -1 });
  res.render("aduan/index", { senaraiAduan, carian });
};
// Cipta aduan baru
exports.store = async (req, res) => {
  const aduan = await Aduan.create(req.body);
  res.redirect(`/aduan/${aduan.id}`);
};
Controllerasync/await + req.query + req.body

Langkah 1–2

Kemaskini dengan method-override.

Borang HTML hanya menyokong GET & POST. Untuk hantar PUT/DELETE, kita "tukar" kaedah:

const methodOverride = require("method-override");
app.use(methodOverride("_method"));
<form method="POST" action="/aduan/123?_method=PUT">
  <!-- medan borang -->
</form>
Update & Delete?_method=PUT atau ?_method=DELETE
Langkah 3

Mesej flash selepas redirect.

const session = require("express-session");
const flash = require("connect-flash");

app.use(session({ secret: process.env.SESSION_SECRET, resave: false, saveUninitialized: false }));
app.use(flash());

app.use((req, res, next) => {
  res.locals.jaya = req.flash("jaya");   // mesej berjaya
  res.locals.gagal = req.flash("gagal"); // mesej gagal
  next();
});

// Dalam controller:
req.flash("jaya", "Aduan berjaya dikemaskini.");
UXMaklum balas sekali papar selepas tindakan
Langkah 5

Papan pemuka dengan statistik.

exports.dashboard = async (req, res) => {
  const jumlah = await Aduan.countDocuments();
  const baru = await Aduan.countDocuments({ status: "Baru" });
  const siasatan = await Aduan.countDocuments({ status: "Dalam Siasatan" });
  const selesai = await Aduan.countDocuments({ status: "Selesai" });
  const terkini = await Aduan.find().sort({ createdAt: -1 }).limit(5);
  res.render("index", { statistik: { jumlah, baru, siasatan, selesai }, terkini });
};
Papan pemukacountDocuments() mengikut status
